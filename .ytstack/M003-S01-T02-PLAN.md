## What this does

Route handler:

```ts
import { Server } from "@modelcontextprotocol/sdk/server/index.js";
import { ListToolsRequestSchema, CallToolRequestSchema } from "@modelcontextprotocol/sdk/types.js";
import { createMcpHttpHandler } from "@/lib/mcp/http-transport";
import { resolveApiKeyFromHeader } from "@/mcp/auth";

export const runtime = "nodejs";

async function withAuth(req: Request): Promise<{ ok: true; userId: string } | { ok: false; res: Response }> {
  const userId = await resolveApiKeyFromHeader(req.headers.get("authorization"));
  if (!userId) {
    return { ok: false, res: new Response(JSON.stringify({ error: "unauthorized" }), {
      status: 401, headers: { "content-type": "application/json", "www-authenticate": 'Bearer realm="sunoflow"' },
    })};
  }
  return { ok: true, userId };
}

function buildServerFor(userId: string): Server {
  const server = new Server({ name: "sunoflow-mcp", version: "0.3.0" }, { capabilities: { tools: {} } });
  server.setRequestHandler(ListToolsRequestSchema, async () => ({
    tools: [{
      name: "sunoflow_info",
      description: "Returns SunoFlow MCP server version + tool list",
      inputSchema: { type: "object", properties: {}, additionalProperties: false },
    }],
  }));
  server.setRequestHandler(CallToolRequestSchema, async (req) => {
    if (req.params.name !== "sunoflow_info") {
      throw new Error(`Unknown tool: ${req.params.name}`);
    }
    return { content: [{ type: "text", text: JSON.stringify({ server: "sunoflow-mcp", version: "0.3.0", userId, tools: ["sunoflow_info"] }, null, 2) }] };
  });
  return server;
}

export async function POST(req: Request): Promise<Response> {
  const auth = await withAuth(req);
  if (!auth.ok) return auth.res;
  const handler = createMcpHttpHandler({ buildServer: () => buildServerFor(auth.userId) });
  return handler(req);
}

export async function GET(req: Request): Promise<Response> {
  const auth = await withAuth(req);
  if (!auth.ok) return auth.res;
  const handler = createMcpHttpHandler({ buildServer: () => buildServerFor(auth.userId) });
  return handler(req);
}
```

Auth refactor (`mcp/auth.ts`):

```ts
async function resolveApiKey(rawKey: string | null | undefined): Promise<string | null> {
  if (!rawKey || !rawKey.startsWith("sk-")) return null;
  const keyHash = createHash("sha256").update(rawKey).digest("hex");
  const apiKey = await prisma.apiKey.findFirst({ where: { keyHash, revokedAt: null }, select: { id: true, userId: true } });
  if (!apiKey) return null;
  prisma.apiKey.update({ where: { id: apiKey.id }, data: { lastUsedAt: new Date() } }).catch(() => {});
  return apiKey.userId;
}

export async function resolveApiKeyFromEnv(): Promise<string | null> {
  return resolveApiKey(process.env.SUNOFLOW_API_KEY);
}

export async function resolveApiKeyFromHeader(authHeader: string | null | undefined): Promise<string | null> {
  if (!authHeader) return null;
  const match = authHeader.match(/^Bearer\s+(.+)$/i);
  if (!match) return null;
  return resolveApiKey(match[1].trim());
}
```
